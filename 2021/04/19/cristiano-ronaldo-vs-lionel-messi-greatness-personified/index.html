Cristiano Ronaldo vs Lionel Messi, Greatness Personified.

2021-04-19

Before the start of the 2003/2004 season, Manchester United signed Cristiano Ronaldo, at the time the most expensive teenager in English Football history,

library(tidyverse)
library(scales)
library(ggrepel)
library(waffle)
Best <- read_csv("goats.csv")
goal <- Best %>%
  group_by(Age, Player) %>%
  summarise(total = sum(Gls))  %>% 
  arrange(Player) %>% 
  group_by(Player) %>% 
  mutate(
   cumulativetotal= cumsum(total)
  )
## `summarise()` has grouped output by 'Age'. You can override using the `.groups` argument.
ggplot() + 
  geom_step(data = goal, aes(x= Age, y=cumulativetotal, group = Player, color = Player)) +
  labs(x="Age", 
       y="Cumulative Goals", 
       title="", 
       subtitle="", 
       caption="Source: Sports-Reference.com | By Samvruth Raj") +  
  scale_color_manual(values = c("orangered", "purple1")) +
  theme_minimal() + 
  theme(
    plot.title = element_text(size = 12, face = "bold"),
    axis.title = element_text(size = 9), 
    plot.subtitle = element_text(size=10)
    )

Comp <- Best %>%
  group_by(Age, Player, CompType) %>%
  summarise(total = sum(Gls))  %>% 
  arrange(Player, CompType) %>% 
  group_by(Player, CompType) %>% 
  mutate(
   cumulativetotal= cumsum(total)
  )
## `summarise()` has grouped output by 'Age', 'Player'. You can override using the `.groups` argument.
ggplot() + 
  geom_step(data = Comp , aes(x= Age, y=cumulativetotal, group = Player, color = Player)) +
  facet_wrap(.~CompType) +
  labs(x="Age", 
       y="Cumulative Goals across competitions", 
       title="", 
       subtitle="", 
       caption="Source: Sports-Reference.com | By Samvruth Raj") +  
  scale_color_manual(values = c("black", "#EDBB00")) +
  theme_minimal() + 
  theme(
    plot.title = element_text(size = 12, face = "bold"),
    axis.title = element_text(size = 9), 
    plot.subtitle = element_text(size=10), 
    )
